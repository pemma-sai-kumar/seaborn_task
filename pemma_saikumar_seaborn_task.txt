import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Load the "tips" dataset
tips= sns.load_dataset("tips")
avg_total_bill = tips.groupby(['day','time'])['total_bill'].mean().reset_index()
avg_total_bill

#Maximum tip amount by smoker status

max_tip = tips.groupby('smoker')['tip'].max().reset_index()
max_tip

# Bar plot for average total bill
plt.figure(figsize=(8,5))
sns.barplot(data=avg_total_bill, x='day', y='total_bill', hue='time')
plt.title("Average Total Bill by Day and Time")
plt.ylabel("Average Total Bill")
plt.xlabel("Day")
plt.legend(title='Time')
plt.tight_layout()
plt.show()
# Bar plot for maximum tip by smoker
plt.figure(figsize=(5,4))
sns.barplot(data=max_tip, x='smoker', y='tip')
plt.title("Maximum Tip by Smoker Status")
plt.ylabel("Max Tip")
plt.xlabel("Smoker")
plt.tight_layout()
plt.show()

Task 2: Penguins Dataset Correlation

# Load the "penguins" dataset
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

penguins = sns.load_dataset("penguins")

#  Select only numerical columns
numeric_penguins = penguins.select_dtypes(include='number')


# Create correlation matrix

corr = numeric_penguins.corr()
print("Correlation Matrix:\n")
print(corr)
print("\n")

# Visualize the correlation heatmap

plt.figure(figsize=(8,6))
sns.heatmap(corr, annot=True, cmap='coolwarm', fmt=".2f")
plt.title("Correlation Heatmap - Penguins Dataset")
plt.show()
corr_pairs = corr.unstack().sort_values(ascending=False)
corr_pairs = corr_pairs[corr_pairs < 1]  # remove self-correlation

top_pair = corr_pairs.idxmax()
top_value = corr_pairs.max()

print(f"Most strongly correlated variables: {top_pair}")
print(f"Correlation value: {top_value:.2f}")
